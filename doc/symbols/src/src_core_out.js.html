  7 YAQP.out = (function() {
  8 	var buffer = [];
  9 	var text = [];
 10 	var objs = [];
 11 	
 12 	var out = {};
 13 	
 14 	var processValue = function(v, o) {
 15 		try {
 16 			switch (typeof v) {
 17 				case "string" :
 18 					return v;
 19 				case "function" :
 20 					var ret = v(o);
 21 					return typeof ret == "string" ? ret + YAQP.game.getBuffer() : YAQP.game.getBuffer();
 22 				default :
 23 					return "";
 24  			};
 25 		} catch(e) {
 26 			YAQP.Functions.error("YAQP.out.processValue", e);
 27 		};
 28 	};
 29 	
 30 	var processText = function() {
 31 		try {
 32 			var here = YAQP.Functions.here();
 33 			text = YAQP.out.formatStr(processValue(here.dsc, here));
 34 		} catch(e) {
 35 			YAQP.Functions.error("YAQP.out.setBuffer", e);
 36 		}; 
 37 	};
 38 	
 39 	var processObjs = function() {
 40 		try {
 41 			objs = [];
 42 			var here = YAQP.Functions.here();
 43 			for (var o in here.objs) {
 44 				if (here.objs.hasOwnProperty(o)){
 45 					var arr = YAQP.out.formatStr(processValue(here.objs[o].dsc, here.objs[o]), here.objs[o]);
 46 					objs = objs.concat(arr);
 47 				}
 48 			};
 49 				
 50 		} catch(e) {
 51 			YAQP.Functions.error("YAQP.out.setBuffer", e);
 52 		}; 
 53 	};
 54 	
 55 	var processBuffer = function() {
 56 		try {
 57 			buffer = YAQP.out.formatStr(YAQP.game.getBuffer());
 58 		} catch(e) {
 59 			YAQP.Functions.error("YAQP.out.setBuffer", e);
 60 		}; 
 61 	};
 62 	
 63 	/**
 64 	 * Возвращает buffer
 65 	 */
 66 	out.getBuffer = function() {
 67 		return buffer;
 68 	};
 69 	
 70 	/**
 71 	 * Возвращает dsc
 72 	 */
 73 	out.getText = function() {
 74 		return text;
 75 	};
 76 	
 77 	/**
 78 	 * Возвращает objs
 79 	 */
 80 	out.getObjs = function() {
 81 		return objs;
 82 	};
 83 
 84 	out.refresh = function() {
 85 		processBuffer();
 86 		processText();
 87 		processObjs();
 88 		
 89 	};
 90 
 91 	return out;
 92 })();

 94 YAQP.out.obj = function(text, obj) {
 95 	this.text = text;
 96 	this.obj = obj;
 97 };

 99 YAQP.out.formatStr = function(s, obj) {
100 	var b_in = s;
101 	var b_out = [];
102 	var pointer = 0;
103 	var pointer_start = pointer;
104 	
105 	var printOutObj = function(){
106 		b_out.push(new YAQP.out.obj(b_in.substring(pointer_start, pointer), obj)); 
107 	}
108 	
109 	var printOut = function(){
110 		var s = b_in.substring(pointer_start, pointer);
111 		if (s != "")
112 			b_out.push(s);
113 		pointer_start = pointer;
114 	};
115 	
116 	var skipChar = function() {
117 		printOut();
118 		pointer_start = pointer + 1;
119 	};
120 	
121 	var printObject = function(obj) {
122 		printOut();
123 		pointer_start = pointer + 1;
124 		for( ; pointer < b_in.length; pointer++) {
125 			if (b_in[pointer] == "}") {
126 				printOutObj();
127 				break;
128 			};
129 		};
130 		pointer++;
131 		pointer_start = pointer;
132 		
133 	};
134 	
135 	for (pointer_start = 0, pointer = 0; pointer < b_in.length; pointer++) {
136 		switch (b_in[pointer]) {
137 			case "\\":
138 				skipChar();
139 				pointer++;
140 				break;
141 			case "{":
142 				printObject();
143 				break;
144 		};
145 	}
146 	printOut();
147 	return b_out;
148 };
