 32 	/**
 33 	 *

 36 		if (sCore != undefined && sGame != undefined) {
 37 			var data = sCore.replace(sQuestEntry, sGame);
 38 			$.globalEval(data);
 39 		}
 40 	}
 41 	/** @scope YAQP */
 42 	return {
 43 		/**
 44 		 * @memberOf YAQP
 45 		 * @param {string}
 46 		 *            corePath Путь к файлу
 47 		 * @param {string}
 48 		 *            gamePath Путь к файлу
 49 		 */

 62 		 * @param {string}
 63 		 *            gamePath Путь к файлу
 64 		 */
