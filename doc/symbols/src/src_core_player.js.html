  5 /**
  6  * Объект, представляющий игрока.
  7  * 
  8  * @constructor
  9  * @class
 10  */
 11 YAQP.Classes.Player = function() {
 12 	/**
 13 	 * Инвентарь.
 14 	 * 
 15 	 * @type YAQP.Classes.ObjList
 16 	 */
 17 	this.inv = new YAQP.Classes.ObjList();
 18 
 19 	this.objs = this.inv;
 20 
 21 	/**
 22 	 * На какой сцене находится игрок.
 23 	 * 
 24 	 * @type YAQP.Classes.Room
 25 	 */
 26 	this.where = undefined;
 27 
 28 	/**
 29 	 * Предыдущая сцена.
 30 	 * 
 31 	 * @type YAQP.Classes.Room
 32 	 */
 33 	this.from = undefined;
 34 
 35 	/**
 36 	 * Перемещает игрока с текущей сцены.
 37 	 * 
 38 	 * @param {YAQP.Classes.Room}
 39 	 *            to Сцена назначения.
 40 	 */
 41 	this.move = function(to) {
 42 		try {
 43 			this.from = this.where;
 44 			this.where = YAQP.Functions.refRoom(to);
 45 		} catch (e) {
 46 			YAQP.Functions.error("YAQP.Classes.Player.move", e);
 47 		}
 48 	};
 49 };
 50 
 51 /**
 52  * Тип объекта.
 53  * 
 54  * @type YAQP.Classes.ObjectTypes
 55  */
 56 YAQP.Classes.Player.prototype.__type = YAQP.Classes.ObjectTypes.Player;
