  6 /**
  7  * Служебная функция, вызывается один раз при создании игры. На вход получает
  8  * список объектов, объявленный в файле игры. Эти объекты преобразуются в
  9  * объекты типа {@link YAQP.Classes.Obj} и сохраняются в глобальном перечне
 10  * объектов игры {@link YAQP.game.objs}.
 11  * 
 12  * @param {object}
 13  *            obj список игровых объектов.
 14  */
 15 YAQP.FunctionsCore.prepareObjs = function(obj) {
 16 	var o = "";
 17 	try {
 18 		YAQP.game.objs = new YAQP.Classes.ObjList();
 19 		for (o in obj)
 20 			YAQP.game.objs.add(new YAQP.Classes.Obj(obj[o], o));
 21 	} catch (e) {
 22 		YAQP.Functions.error(
 23 				"YAQP.Functions.prepareObjs " + "obj: '" + o + "'", e);
 24 	};
 25 };
 26 
 27 /**
 28  * Служебная функция, вызывается один раз при создании игры. На вход получает
 29  * список сцен, объявленный в файле игры. Эти сцены преобразуются в объекты типа
 30  * {@link YAQP.Classes.Room} и сохраняются в глобальном перечне сцен игры
 31  * {@link YAQP.game.rooms}.
 32  * 
 33  * @param {object}
 34  *            room список сцен.
 35  */
 36 YAQP.FunctionsCore.prepareRooms = function(room) {
 37 	var r = "";
 38 	try {
 39 		YAQP.game.rooms = new YAQP.Classes.RoomList();
 40 		for (r in room)
 41 			YAQP.game.rooms.add(new YAQP.Classes.Room(room[r], r));
 42 	} catch (e) {
 43 		YAQP.Functions.error("YAQP.Functions.prepareRooms " + "room: '" + r
 44 						+ "'", e);
 45 	}
 46 };

 48 /**
 49  * Служебная функция, вызывается один раз при создании игры. На вход получает
 50  * список сцен, объявленный в файле игры. Инициализирует список объектов сцены.
 51  * 
 52  * @TODO написать нормальную документацию.
 53  * @param {object}
 54  *            room список сцен.
 55  */
 56 YAQP.FunctionsCore.prepareLinksRoomObjs = function(room) {
 57 	var r = "";
 58 	var o = "";
 59 	try {
 60 		for (var r in room) {
 61 			for (var o in room[r].obj) {
 62 				YAQP.Functions.put(room[r].obj[o], YAQP.game.rooms[r]);
 63 			}
 64 		}
 65 	} catch (e) {
 66 		YAQP.Functions.error("YAQP.Functions.prepareLinksRoomObjs " + "room: '"
 67 						+ r + "', obj: '" + o + "' " + "room[r].obj[o]: '"
 68 						+ YAQP.Functions.toString(room[r].obj[o]) + "'", e);
 69 	}
 70 };

 72 /**
 73  * Служебная функция, вызывается один раз при создании игры. На вход получает
 74  * список сцен, объявленный в файле игры. Инициализирует список возможных
 75  * переходов со сцены.
 76  * 
 77  * @param {object}
 78  *            room список сцен.
 79  */
 80 YAQP.FunctionsCore.prepareLinksRoomWays = function(room) {
 81 	var r = "";
 82 	var w = "";
 83 	try {
 84 		for (r in room) {
 85 			for (w in room[r].way) {
 86 				YAQP.game.rooms[r].ways.add(room[r].way[w]);
 87 			}
 88 		}
 89 	} catch (e) {
 90 		YAQP.Functions.error("YAQP.Functions.prepareLinksRoomWays " + "room: '"
 91 						+ r + "', way: '" + w + "' " + "room[r].way[w]: '"
 92 						+ YAQP.Functions.toString(room[r].way[w]) + "'", e);
 93 	}
 94 };

 98  * список предметов, объявленный в файле игры. Инициализирует список связанных

116 						+ YAQP.Functions.toString(obj[o].obj[oo]) + "'", e);

118 	}
119 };

125  * @param fieldName
126  *            Название поля.
127  * @returns {Boolean} результат проверки.
128  */
129 YAQP.Functions.isValidRoomField = function(fieldName) {
130 	switch (fieldName) {
131 		case "objs" :
132 		case "ways" :
133 		case "__type" :
134 			return false;
135 		default :
136 			return true;
137 	}
