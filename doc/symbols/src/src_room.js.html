  5 /**
  6  * Создает новую комнату. На вход получает описание комнаты и ее 
  7  * идентификатор. 
  8  * @constructor
  9  * @class
 10  * @param {string}
 11  *            loc Путь к файлу
 12  * @param {string}
 13  *            id Путь к файлу
 14  */
 15 YAQP.Classes.Room = function(loc, id) {
 16 	/**
 17 	 * Название сцены. Это название будет заголовком сцены
 18 	 * при её отображении. Имя сцены также используется 
 19 	 * для её идентификации при переходах. 
 20  	 * @type string
 21 	 */
 22 	this.name = "";
 23 	/**
 24 	 * Описание комнаты. Описание статической части 
 25 	 * сцены, которое выводится при входе в сцену или 
 26 	 * выполнении команды look
 27 	 * @type string
 28 	 */
 29 	this.dsc = "";
 30 	/**
 31 	 * Определяет, выводить описание статической части сцены на 
 32 	 * каждом ходу (значение true) или только при первом входе на 
 33 	 * сцену (значение false). По-умолчанию установлено false.
 34 	 * @type boolean
 35 	 */
 36 	this.forcedsc = false;
 37 	/**
 38 	 * Список объектов сцены. 
 39 	 * @type YAQP.Classes.ObjectList
 40 	 */
 41 	this.obj = new YAQP.Classes.ObjectList();
 42 
 43 	for (var p in loc) {
 44 		if (p != "obj") {
 45 			this[p] = loc[p];
 46 		}
 47 	}
 48 	
 49 	for (var i = 0; i < loc.obj.length; i++) {
 50 		if (typeof loc.obj[i] == "string") {
 51 			loc.obj[i] = obj[loc.obj[i]];
 52 		}
 53 	}
 54 	if (this.obj == undefined) {
 55 		this.obj = [];
 56 	}
 57 	this.id = id;
 58 };
 59 
 60 YAQP.Classes.Room.prototype.srch = function() {
 61 	var obj = this;
 62 	var s;
 63 	switch (typeof obj) {
 64 		case "object" :
 65 			s = obj.id;
 66 			break;
 67 		case "string" :
 68 			s = obj;
 69 			break;
 70 		default :
 71 			return undefined;
 72 			break;
 73 	}
 74 
 75 	for (var arrVal in this.loc) {
 76 		if (arrVal != undefined) {
 77 			if (arrVal.id == s) {
 78 				return arrVal;
 79 			}
 80 		}
 81 	}
 82 };
