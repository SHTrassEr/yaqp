 20 /**
 21  * Добавляет объект в список. Если в качестве параметра указана строка, то
 22  * объект выбирается из глобального списка обьектов.
 23  * 
 24  * @param {YAQP.Classes.Object|string}
 25  *            o объект поиска. Либо строка с id объекта, либо сам объект.
 26  */
 27 YAQP.Classes.ObjectList.prototype.add = function(o) {
 28 	try {
 29 		if (typeof o === "object") {
 30 			if (o.__id) {
 31 				if (this[o.__id] !== o) {
 32 					this[o.__id] = o;
 33 					return true;
 34 				} else
 35 					/* если объект уже занесен в список */
 36 					return false;
 37 			} else
 38 				throw "Не указано свойство __id, o: '" + o.toString() + "'";
 39 		} else
 40 			throw "Параметр не является объектом, o: '" + o + "'";
 41 		return false;
 42 	} catch (e) {
 43 		YAQP.Functions.error("YAQP.Classes.ObjectList.add ", e);
 44 	}
 45 };

 47 /**
 48  * Поиск объекта. Функция возвращает найденный в списке объект.
 49  * 
 50  * @param {YAQP.Classes.Object|string}
 51  *            o объект поиска. Либо строка с id объекта, либо сам объект.
 52  * @param {boolean}
 53  *            lookHidden Искать скрытые объекты или нет. По умолчанию скрытые
 54  *            объекты не ищутся.
 55  * @returns {number|undefined} возвращает индекс найденного объекта, либо
 56  *          undefined если объект не найден.
 57  */
 58 YAQP.Classes.ObjectList.prototype.srch = function(o, lookHidden) {
 59 	switch (typeof o) {
 60 		case "string" :
 61 			if (this[o]) {
 62 				if (lookHidden || this[o].isEnabled())
 63 					return this[o];
 64 			}
 65 			break;
 66 		case "object" :
 67 			if (this[o.__id]) {
 68 				if (lookHidden || this[o.__id].isEnabled())
 69 					if (this[o.__id] === o)
 70 						return this[o.__id];
 71 			}
 72 			break;
 73 	}
 74 	return undefined;
 75 };

 77 /**
 78  * Скрытие объекта в списке; отличается от удаления тем, что может быть
 79  * возвращён к жизни через метод enable
 80  * 
 81  * @param {YAQP.Classes.Object|string}
 82  *            o объект. Либо строка с id объекта, либо сам объект.
 83  * @returns {number|undefined} возвращает false, если объект удалось скрыть,
 84  *          либо undefined, если объект не найден.
 85  */
 86 YAQP.Classes.ObjectList.prototype.disable = function(o) {
 87 	var obj = this.srch(o);
 88 	if (obj !== undefined) {
 89 		return obj.disable();
 90 	}
 91 	return false;
 92 };

111 /**
112  * Удаление объекта из списка. Скрытые объекты не удаляются.
113  * 
114  * @param {YAQP.Classes.Object|string}
115  *            o объект. Либо строка с id объекта, либо сам объект.
116  * @returns {boolean} возвращает true, если объект удалось удалить,
117  *          либо false, если объект не найден.
118  */
119 YAQP.Classes.ObjectList.prototype.del = function(o) {
120 	var obj = this.srch(o);
121 	if (obj !== undefined) {
122 		delete this[obj.__id];
123 		return true;
124 	}
125 	return false;
126 };
127 /**
128  * Удаление объекта из списка. Скрытые объекты тоже удаляются.
129  * 
130  * @param {YAQP.Classes.Object|string}
131  *            o объект. Либо строка с id объекта, либо сам объект.

133  *          либо undefined, если объект не найден.

135 YAQP.Classes.ObjectList.prototype.purge = function(o) {

137 	if (obj !== undefined) {
138 		delete this[obj.__id];
139 		return true;
140 	}
141 	return false;

144 /**
145  * Замена объекта в списке. Если объект, на который следует заменить (oNew) уже
146  * есть в списке, то просто удаляется объект замены (oOld). Если исходный 
147  * объект не найден, то список остается без изменений; 
148  * 
149  * @param {YAQP.Classes.Object|string}
150  *            oOld объект поиска. Либо строка с id объекта, либо сам объект.
151  * @param {YAQP.Classes.Object|string}
152  *            oNew объект поиска. Либо строка с id объекта, либо сам объект.
153  * @returns {boolean} возвращает true, если замена произошла, false - в
154  *          противном случае.
155  */
156 YAQP.Classes.ObjectList.prototype.replace = function(oOld, oNew) {
157 	try {
158 		if (typeof oOld != "object") {
159 			return false;
160 		}
161 		if (typeof oNew != "object") {
162 			return false;
163 		}
164 		if (oOld === oNew)
165 			return true;
166 
167 		if (this.del(oOld)) {
168 			this.add(oNew);
169 			return true;
170 		} else
171 			return false;
172 
173 	} catch (e) {
174 		YAQP.Functions.error("YAQP.Classes.ObjectList.replace ", e);
175 	};
176 };
