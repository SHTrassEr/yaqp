  7  * 
  8  * @param {object}
  9  *            Child Наследник.
 10  * @param {object}

 13 YAQP.Functions.extend = function(Child, Parent) {
 14 	var F = function() {
 15 	};

 17 	Child.prototype = new F();
 18 	Child.prototype.constructor = Child;
 19 	Child.superclass = Parent.prototype;

 22 /**
 23  * Функция обработки ошибок.
 24  * 
 25  * @param {string} msg сообщение об ошибки.
 26  * @param {string|Error} Дополнительная информация об ошибке.
 27  */
 28 YAQP.Functions.error = function(msg, e) {
 29 	switch (typeof e) {
 30 		case "undefined" :
 31 			throw new Error(msg + "\n");
 32 			break;
 33 		case "string" :
 34 			throw new Error(msg + e + "\n");
 35 			break;
 36 		case "object" :
 37 			throw new Error(msg + "\n" + e.message);
 38 			break;
 39 	}
 40 };

246 /**
247  * Получает комнату. Параметром этой функции может быть либо комната либо 
248  * идентификатор комнаты. Особенность функции в том, что она либо вернет комнату,
249  * либо сгенерирует ошибку. Если в качестве параметра передан объект, то
250  * проверяется, является ли этот объект комнатой.
251  * 
252  * @param {YAQP.Classes.Room|string} name название комнаты, идентификатор либо сама комната.
253  * @returns {YAQP.Classes.Room} Ссылка на искомую комнату.
254  */
255 YAQP.Functions.refRoom = function(name) {
256 	try {
257 		switch (typeof name) {
258 			case "string" : /** если строка - то ищем в глобальном списке комнат */
259 				if (YAQP.game.rooms.srch(name)) {
260 					return YAQP.game.rooms.srch(name);
261 				} else
262 					throw "Объект не является комнатой name : " + 
263 						YAQP.Functions.quotes(name);
264 				break;
265 			case "object" : /** если объект - то проверяем, является ли он комнатой */
266 				if (YAQP.Functions.isRoom(name)) {
267 					return name;
268 				} else
269 					throw "Объект не является комнатой name : "
270 							+ YAQP.Functions.quotes(name);
271 				break;
272 			default :
273 				throw "Объект не является комнатой name : "
274 						+ YAQP.Functions.quotes(name);
275 				break;
276 		};
277 	} catch (e) {
278 		YAQP.Functions.error("YAQP.Functions.refRoom", e);
279 	};
280 };

314 /**
315  * Получает игровой предмет или комнату. Использует функции {@link YAQP.Functions.refObj} и
316  * {@link YAQP.Functions.refRoom}. Если переданный параметр не является ни комнатой ни 
317  * игровым предметом, то генерируется ошибка. На данный момент нигде не используется, так как 
318  * всегда из контекста ясно, что мы хотим найти: предмет или комнату. Возможно, что 
319  * эта функция в дальнейшем будет удалена. Иди же оставлена для совместимости со STEAD.  
320  * @param {YAQP.Classes.obj|string} name название предмета/комнаты, идентификатор либо предмет/комната.
321  * @returns {YAQP.Classes.obj} Ссылка на искомый игровой предмет/комнату.
322  */
323 
324 YAQP.Functions.ref = function(name) {
325 	try {
326 		switch (typeof name) {
327 			case "string" :
328 				if (YAQP.game.rooms.srch(name))
329 					return YAQP.game.rooms.srch(name);
330 				if (YAQP.game.objs.srch(name))
331 					return YAQP.game.objs.srch(name);
332 				break;
333 			case "object" :
334 				if (YAQP.Functions.isRoom(name) || YAQP.Functions.isObj(name)) {
335 					return name;
336 				} else
337 					throw "Объект не является ни комнатой ни предметом o : "
338 							+ YAQP.Functions.quotes(name);
339 				break;
340 		}
341 		throw "Объект не является ни комнатой ни предметом o : "
342 			+ YAQP.Functions.quotes(name);
343 	} catch (e) {
344 		YAQP.Functions.error("YAQP.Functions.ref", e);
345 	};
346 };

348 /**
349  * Возвращает список объектов указанной комнаты; если комната не указана, то
350  * возвращает список объектов текущей комнаты. Текущая комната запрашиваетя через функцию 
351  * {@link YAQP.Functions.here}. Если же комната указана, то ссылка на нее запрашивается
352  * через функцию {@link YAQP.Functions.refRoom}. Эта функция либо возвращает список
353  * объектов (класс {@link YAQP.Classes.ObjList}), либо генерирует ошибку. 
354  * 
355  * @param {YAQP.Classes.Room|string} room название комнаты, идентификатор либо сама комната.
356  * 
357  * @returns {YAQP.Classes.ObjList} Объекты сцены.
358  */
359 YAQP.Functions.objs = function(room) {
360 	try {
361 		if (room === undefined) {
362 			if (YAQP.Functions.isObjList(YAQP.Functions.here().objs)) {
363 				return YAQP.Functions.here().objs;
364 			} else
365 				throw "В текущей сцене найден не инициализированный"
366 						+ "список предметов YAQP.Functions.here() : "
367 						+ YAQP.Functions.quotes(YAQP.Functions.here())
368 						+ " objs : "
369 						+ YAQP.Functions
370 								.quotes(YAQP.Functions.here().objs);
371 
372 		} else {
373 			var r = YAQP.Functions.refRoom(room); 
374 			if (YAQP.Functions.isObjList(r.objs)) {
375 				return r.objs;
376 			} else {
377 				throw "В переданной сцене найден не инициализированный"
378 						+ "список объектов room : " + YAQP.Functions.quotes(room)
379 						+ " objs : " + YAQP.Functions.quotes(r.objs);
380 
381 			}
382 		} 
383 	} catch (e) {
384 		YAQP.Functions.error("YAQP.Functions.objs", e);
385 	};
386 };

427 /**
428  * Удаляет предмет из указанной комнаты или из инвентаря игрока; если комната не указана, то
429  * удаляет предмет из текущей комнаты. Текущая комната запрашиваетя через функцию 
430  * {@link YAQP.Functions.here}. Если же комната указана, то ссылка на нее запрашивается
431  * через функцию {@link YAQP.Functions.refRoom}. Ссылка на объект запрашивается через функцию 
432  * {@link YAQP.Functions.refObj}. Если объект удаляется из комнаты, то 
433  * его {@link YAQP.Classes.Obj.where} становится равным undefined, если же нет --- то остается
434  * без изменений.
435  * 
436  * @param {YAQP.Classes.Obj|string} obj название предмета, идентификатор либо сам предмет.
437  * @param {YAQP.Classes.Room|string} room название комнаты, идентификатор либо сама комната.
438  * @returns {boolean} true, если объект удалось удалить, false в противном случае.
439  */
440 YAQP.Functions.remove = function(obj, room) {
441 	try {
442 		var o = YAQP.Functions.refObj(obj);
443 		if (room === undefined) {
444 			
445 		} else if (YAQP.Functions.isPlayer(room)) {
446 			if (room.objs.del(o)) {
447 				o.where = undefined
448 				return true;
449 			}
450 		} else {
451 			var r = YAQP.Functions.refRoom(room);
452 			if (YAQP.Functions.isObjList(r.objs)) {
453 				if (r.objs.del(o)) {
454 					o.where = undefined
455 					return true;
456 				}
457 			} else
458 				throw "В конмнате не инициализирован скисок объектов: room : '"
459 						+ YAQP.Functions.toString(room) + "', obj : '"
460 						+ YAQP.Functions.toString(obj) + "'";
461 		}
462 		return false;
463 	} catch (e) {
464 		YAQP.Functions.error("YAQP.Functions.remove", e);
465 	}
466 };

468 /**
469  * Заменяет один объект в указанной комнате на другой; если комната не указана,
470  * то замена производится в текущей комнате. Текущая комната запрашиваетя через функцию 
471  * {@link YAQP.Functions.here}. Если комната указана, то ссылка на нее запрашивается
472  * через функцию {@link YAQP.Functions.refRoom}. Ссылки на объекты запрашиваются через функцию 
473  * {@link YAQP.Functions.refObj}. Если объект-источник не найден в списке, то список
474  * остается без изменений. Если объект-источник удаляется из комнаты, то 
475  * его {@link YAQP.Classes.Obj.where} становится равным undefined. 
476  * {@link YAQP.Classes.Obj.where} объекта для замены становится равным комнате.
477  * 
478  * @param {YAQP.Classes.Obj|string} objSrc название предмета-источника, 
479  * 	идентификатор либо сам предмет.
480  * @param {YAQP.Classes.Obj|string} objDst название предмета для замены, 
481  * 	идентификатор либо сам предмет.
482  * @param {YAQP.Classes.Room|YAQP.Classes.Player|string} room название комнаты, 
483  * идентификатор, сама комната либо ссылка на игрока.
484  */
485 YAQP.Functions.replace = function(objSrc, objDst, room) {
486 	try {
487 		var list;

493 		} else
494 			list = YAQP.Functions.refRoom(room).objs;
495 		if (list.replace(os, od)) {
496 			os.where = undefined;
497 			return true;	
498 		} else
499 			return false;
500 	} catch (e) {
501 		YAQP.Functions.error("YAQP.Functions.replace", e);
502 	}

559 /**
560  * Переход из одной комнаты в другую.
561  * 
562  * @param {YAQP.Classes.Room|string} название комнаты или сама 
563  * комната в которую следует перейти.
564  * @param {boolean} устанавливается в true если вызывается при опсании игры
565  */
566 YAQP.Functions.go = function(to, ingame) {
567 	try {
568 		if (!ingame)
569 			YAQP.game.go_canceled = false;
570 		var room_from = YAQP.Functions.here();
571 		var room_to = YAQP.Functions.refRoom(to);
572 		
573 		if (!ingame)
574 			room_from.exit(room_from, room_to);
575 		if (YAQP.game.go_canceled) {
576 			return;
577 		}
578 		room_to.enter(room_to, room_from);
579 		if (YAQP.game.go_canceled) {
580 			return;
581 		}
582 		
583 		YAQP.Functions.me().move(room_to);
584 		
585 		if (!ingame)
586 			room_from.leave(room_from, room_to);
587 		room_to.entered(room_to, room_from);
588 		
589 	} catch (e) {
590 		YAQP.Functions.error("YAQP.Functions.go", e);
591 	};
592 };

669 	try {
670 		switch (typeof o) {
671 			case "string" :
672 				var oo = YAQP.game.objs.srch(o);

677 							+ YAQP.Functions.toString(o) + "'";
678 				break;

681 					return o.where;

683 					throw "Параметр не является предметом.";
684 				break;
685 		};
686 	} catch (e) {
687 		YAQP.Functions.from("YAQP.Functions.where", e);
688 	};
