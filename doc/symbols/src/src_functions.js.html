  5 /**
  6  * Функция делает класс Child потомком класса Parent.
  7  * 
  8  * @param {object}
  9  *            Child Наследник.
 10  * @param {object}
 11  *            Parent Родитель.
 12  */
 13 YAQP.Functions.extend = function(Child, Parent) {
 14 	var F = function() {
 15 	};
 16 	F.prototype = Parent.prototype;
 17 	Child.prototype = new F();
 18 	Child.prototype.constructor = Child;
 19 	Child.superclass = Parent.prototype;
 20 };

 22 /**
 23  * Проверяет, является ли переданный объект комнатой.
 24  * 
 25  * @param {YAQP.Classes.Room}
 26  *            комната.
 27  * @returns {boolean} Возвращает true, если объект является комнатой. Возвращает
 28  *          false если объект комнатой не является.
 29  */
 30 YAQP.Functions.isRoom = function(o) {
 31 	if (typeof o == "object")
 32 		if (o.__type == YAQP.Classes.ObjectTypes.Room)
 33 			return true;
 34 	return false;
 35 };

 61 /**
 62  * Возвращает текущую сцену.
 63  * 
 64  * @returns {YAQP.Classes.ObjList} Инвентарь.
 65  */
 66 YAQP.Functions.here = function() {
 67 	if (YAQP.Functions.isRoom(YAQP.game.pl.where))
 68 		return YAQP.game.pl.where;
 69 	throw new Error("Не удалось определить местоположение игрока.");
 70 };

 72 /**
 73  * Возвращает список объектов указанной сцены; если сцена не указана, то
 74  * возвращает список объектов текущей сцены
 75  * 
 76  * @returns {YAQP.Classes.ObjList} Объекты сцены.
 77  */
 78 YAQP.Functions.objs = function(o) {
 79 	if (o === undefined) {
 80 		o = YAQP.Functions.here();
 81 	}
 82 	if (YAQP.Functions.isRoom(o))
 83 		return o.objs;
 84 };

 86 /**
 87  * Служебная функция, вызывается один раз при создании игры. 
 88  * На вход получает список объектов, объявленный в файле 
 89  * игры. Эти объекты преобразуются в объекты типа 
 90  * {@link YAQP.Classes.Obj} и сохраняются в глобальном 
 91  * перечне объектов игры {@link YAQP.game.objs}. 
 92  * @param {object} obj список игровых объектов.
 93  */
 94 YAQP.Functions.prepareObjs = function(obj) {
 95 	YAQP.game.objs = {};
 96 	for (var o in obj)
 97 		YAQP.game.objs[o] = new YAQP.Classes.Obj(obj[o], o);
 98 };

100 /**
101  * Служебная функция, вызывается один раз при создании игры. 
102  * На вход получает список сцен, объявленный в файле 
103  * игры. Эти сцены преобразуются в объекты типа 
104  * {@link YAQP.Classes.Room} и сохраняются в глобальном 
105  * перечне сцен игры {@link YAQP.game.rooms}. 
106  * @param {object} room список сцен.
107  */
108 YAQP.Functions.prepareRooms = function(room) {
109 	YAQP.game.rooms = {};
110 	for (var r in room) 
111 		YAQP.game.rooms[r] = new YAQP.Classes.Room(room[r], r);
112 };
