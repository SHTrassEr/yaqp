  5 /**
  6  * Список объектов типа YAQP.Classes.Obj. Класс YAQP.Classes.ObjList наследутеся
  7  * от YAQP.Classes.ObjectList
  8  * 
  9  * @constructor
 10  * @class
 11  */
 12 YAQP.Classes.ObjList = function() {
 13 
 14 };
 15 
 16 /**
 17  * Наследуемся от YAQP.Classes.ObjectList.
 18  */
 19 YAQP.Functions.extend(YAQP.Classes.ObjList, YAQP.Classes.ObjectList);
 20 
 21 /**
 22  * Переопределяем метод добавления обьекта в список. В список могут попасть
 23  * только обьекты типа YAQP.Classes.ObjectTypes.Obj. Добавляет объект в список.
 24  * Если в качестве параметра указана строка, то объект выбирается из глобального
 25  * списка обьектов (game.obj).
 26  * 
 27  * @param {YAQP.Classes.Object|string}
 28  *            o объект поиска. Либо строка с id объекта, либо сам объект.
 29  */
 30 YAQP.Classes.ObjList.prototype.add = function(o) {
 31 	try {
 32 		switch (typeof o) {
 33 			case "string" :
 34 			case "object" :
 35 				return YAQP.Classes.ObjList.superclass.add.apply(this,
 36 						[YAQP.Functions.refObj(o)]);
 37 			default :
 38 				throw "Объект не является игровым предметом o : "
 39 					+ YAQP.Functions.quotes(o);
 40 		}
 41 	} catch (e) {
 42 		YAQP.Functions.error("YAQP.Classes.ObjList.add ", e);
 43 	};
 44 };
